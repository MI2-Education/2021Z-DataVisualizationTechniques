import numpy as np
import pandas as pd
import seaborn as sns

frontalSinusData = pd.DataFrame(np.array([[10, "male", "Symmetry"], 
                                         [12, "female", "Symmetry"], 
                                         [14, "male", "Left assymetry"], 
                                         [4, "female", "Left assymetry"], 
                                         [4, "male", "Right assymetry"], 
                                         [11, "female", "Right assymetry"]]),
                                columns = ["number", "gender", "pattern"])

barplot = sns.barplot(x = "pattern", y = "number", data = frontalSinusData, hue = "gender")
barplot.set(xlabel = "", ylabel = "Number of cases")
